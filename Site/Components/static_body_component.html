StaticBodyComponent

type StaticBodyComponent struct {
	b2Body           *box2d.B2Body
	OwnerEntId       EntId
	OnCollisionBegin ChaiEvent1[CollisionBox2D]
	OnCollisionEnd   ChaiEvent1[CollisionBox2D]
	settings         StaticBodySettings
}

A Static Body is a Physics Body that does not get affected by physical forces at all, And it is by itself an effector.

StaticBodyComponent is a Component that adds Physical properties to an Entity. It basically registers this Entity as part of the Physics Engine. Using this Component you can create boundaries(walls), that prevent DynamicBodies from passing through them.

b2Body: Private variable referring to the pointer of the body in the physics Engine. Unless you're contributing to Chai, this should be of no concern to you.

OwnerEntId: The owner of this dynamic body (EntId is just a wrapper around ecs.Id).

OnCollisionBegin: A ChaiEvent that gets triggered whenever a collision happens on this body (See ChaiEvent).

OnCollisionEnd: A ChaiEvent that gets triggered whenever a collision ends on this body.

settings: Settings concerning this dynamic body (mass, elasticity, locked rotation, etc...)

Create a StaticBodyComponent

Just like previously mentioned, Components authored by Chai follow the convention of custom Components creation methods structured (New + ComponentName).

    _entity := _this_scene.NewEntityId()
    _vt := chai.VisualTransform{
        Position: chai.NewVector2f(0.0, 0.0),
        Z: 0.0,
        Dimensions: chai.NewVector2f(100.0, 100.0),
        Rotation: 0.0,
        Scale: 1.0,
        Tint: chai.WHITE,
        UV1: chai.Vector2fZero,
        UV2: chai.Vector2fOne,
    }
    
    _sb_settings := chai.StaticBodySettings{
        IsTrigger: false,
        ColliderShape: chai.SHAPE_RECTBODY,
        Offset: chai.Vector2fZero,
        StartDimensions: _vt.Dimensions,
        StartRotation: _vt.Rotation,
        Friction: 0.3, Elasticity: 0.25,
        PhysicsLayer: chai.PHYSICS_LAYER_ALL,
    }

    _sb_component := chai.NewStaticBody(_entity, _vt, &_sb_settings)
    
    _this_scene.AddComponents(_entity, chai.ToComponent(_vt), chai.ToComponent(_sb_component))

    _this_scene.NewUpdateSystem(chai.DynamicBodySystem)

Let's discuss "StaticBodySettings"

type StaticBodySettings struct {
	IsTrigger            bool
	ColliderShape        int
	Offset               Vector2f
	StartDimensions      Vector2f
	StartRotation        float32
	Friction, Elasticity float32
	PhysicsLayer         uint16
}

IsTrigger: Whether the body is trigger (passsable) or not

ColliderShape: There are 2 shapes currently (SHAPE_RECTBODY, SHAPE_CIRCLEBODY)

Offset: A position offset that can be modified in runtime

StartDimensions: The body's starting dimensions

StartRotation: The body's starting rotation

Friction: The body's friction with other body's

Elasticity: The body's bounciness

PhysicsLayer: The body's interaction physical layer

Usage Example

func SceneStartSystem(_this_scene *chai.Scene) {
    _db_entity := _this_scene.NewEntityId()
    _db_vt := chai.VisualTransform{
        Position: chai.NewVector2f(0.0, 0.0),
        Z: 0.0,
        Dimensions: chai.NewVector2f(100.0, 100.0),
        Rotation: 0.0,
        Scale: 1.0,
        Tint: chai.WHITE,
        UV1: chai.Vector2fZero,
        UV2: chai.Vector2fOne,
    }
    _texture := chai.LoadPng("chai_logo.png", &chai.TextureSettings{
        Filter: chai.TEXTURE_FILTER_LINEAR,
    })
    _db_sprite_component := chai.NewSpriteComponent(_this_scene, _db_entity, _db_vt, &_texture, false)

    _db_settings := chai.DynamicBodySettings{
        IsTrigger: false,
        ColliderShape: chai.SHAPE_RECTBODY,
        StartPosition: _db_vt.Position,
        Offset: chai.Vector2fZero,
        StartDimensions: _db_vt.Dimensions,
        StartRotation: _db_vt.Rotation,
        Mass: 20, Friction: 0.3, Elasticity: 0.25,
        ConstrainRotation: false,
        PhysicsLayer: chai.PHYSICS_LAYER_ALL,
    }
    _db_component := chai.NewDynamicBodyComponent(_db_entity, _db_vt, &_db_settings)

    _this_scene.AddComponents(_db_entity, chai.ToComponent(_db_vt), chai.ToComponent(_db_sprite_component), chai.ToComponent(_db_component))
    
    _sb_entity := _this_scene.NewEntityId()
    _sb_vt := chai.VisualTransform{
        Position: chai.NewVector2f(0.0, -200.0),
        Z: 0.0,
        Dimensions: chai.NewVector2f(400.0, 50.0),
        Rotation: 0.0,
        Scale: 1.0,
        Tint: chai.WHITE,
        UV1: chai.Vector2fZero,
        UV2: chai.Vector2fOne,
    }
    
    _sb_settings := chai.StaticBodySettings{
        IsTrigger: false,
        ColliderShape: chai.SHAPE_RECTBODY,
        Offset: chai.Vector2fZero,
        StartDimensions: _sb_vt.Dimensions,
        StartRotation: _sb_vt.Rotation,
        Friction: 0.3, Elasticity: 0.25,
        PhysicsLayer: chai.PHYSICS_LAYER_ALL,
    }

    _sb_component := chai.NewStaticBodyComponent(_sb_entity, _sb_vt, &_sb_settings)
    
    _this_scene.AddComponents(_sb_entity, chai.ToComponent(_sb_vt), chai.ToComponent(_sb_component))


    _this_scene.NewUpdateSystem(chai.DynamicBodySystem)
    _this_scene.NewUpdateSystem(chai.DebugBodyDrawSystem)
}

var app chai.App = chai.App{
	Title:  "Docs",
	Width:  480,
	Height: 480,
	OnStart: func() {
		_test_scene := chai.NewScene()
		_test_scene.NewStartSystem(SceneStartSystem)

		// Transition into this scene
		chai.ChangeScene(&_test_scene)
	},
}

func main(){
	chai.Run(&app)
}
